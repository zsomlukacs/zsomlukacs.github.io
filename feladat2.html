#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int t;
    cin >> t;
    while(t--){
        string b1, b2;
        int K;
        cin >> b1 >> b2 >> K;
        int n = b1.size(), m = b2.size();

        const int INF = 1e9;
        vector<vector<vector<int>>> dp(n+1, vector<vector<int>>(m+1, vector<int>(K+1, -INF)));

        // null hosszú subsequence happiness = 0
        for(int i=0;i<=n;i++)
            for(int j=0;j<=m;j++)
                dp[i][j][0] = 0;

        for(int i=1;i<=n;i++){
            for(int j=1;j<=m;j++){
                for(int k=1;k<=K;k++){
                    // ha egyeznek a karakterek
                    if(b1[i-1] == b2[j-1]){
                        if(dp[i-1][j-1][k-1] != -INF)
                            dp[i][j][k] = max(dp[i][j][k], dp[i-1][j-1][k-1] + (int)b1[i-1]);
                    }
                    // kihagyás mindkét irányban
                    dp[i][j][k] = max(dp[i][j][k], dp[i-1][j][k]);
                    dp[i][j][k] = max(dp[i][j][k], dp[i][j-1][k]);
                }
            }
        }

        int ans = dp[n][m][K];
        if(ans == -INF) ans = 0; // nincs K hosszú subsequence
        cout << ans << "\n";
    }
    return 0;
}
