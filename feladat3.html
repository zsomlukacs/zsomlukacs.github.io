C++ Kód TREEDEGREE - Degree of a Tree



#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int T;
    cin >> T;
    while(T--){
        int N;
        cin >> N;
        vector<int> arr(2*N);
        for(int i=0;i<2*N;i++) cin >> arr[i];

        vector<int> degree(N+1,0);
        stack<int> st;

        for(int x : arr){
            if(!st.empty() && st.top() == x){
                st.pop(); // csúcs vége
            } else {
                if(!st.empty()){
                    degree[st.top()]++; // szülőnek nő a gyerekszám
                    degree[x]++; // gyereknek nő a szülőhöz kapcsolódó él
                }
                st.push(x);
            }
        }

        int ans = *max_element(degree.begin()+1, degree.end());
        cout << ans << "\n";
    }

    return 0;
}
