#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
const ll INF = 1e18;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, M, Q;
    cin >> N >> M >> Q;
    vector<vector<pair<int,int>>> adj(N);

    for(int i=0;i<M;i++){ 
        int A,B,L;
        cin >> A >> B >> L;
        adj[A].push_back({B,L});
        adj[B].push_back({A,L});
    }

    vector<int> queries(Q);
    for(int i=0;i<Q;i++) cin >> queries[i];

    for(int qi : queries){
        vector<ll> dist(N, INF);
        priority_queue<pair<ll,int>, vector<pair<ll,int>>, greater<>> pq;

        dist[qi] = 0;
        pq.push({0, qi});

        while(!pq.empty()){
            auto [d,u] = pq.top(); pq.pop();
            if(d > dist[u]) continue;

            for(auto &[v,w]: adj[u]){
                if(dist[v] > dist[u]+w){
                    dist[v] = dist[u]+w;
                    pq.push({dist[v], v});
                }
            }
        }

        ll maxDist = 0;
        int count = 0;
        for(ll d : dist){
            if(d==INF) continue;
            if(d > maxDist){
                maxDist = d;
                count = 1;
            } else if(d == maxDist) count++;
        }

        cout << maxDist << " " << count << "\n";
    }

    return 0;
}
